void shellCommandCall() {
	int fd = open("output.txt", O_WRONLY | O_CREAT | O_TRUNC, 0666);

	if (fd == -1) {
		perror("open");
		return;
	}

	dup2(fd, STDOUT_FILENO); // Redirect stdout to the file
	close(fd);

	execlp("/bin/cat", "cat", "input.txt", NULL);

	// If execlp fails
	perror("execlp");
}
