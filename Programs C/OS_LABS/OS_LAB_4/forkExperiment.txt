#include <stdio.h>
#include <fcntl.h>
#include <unistd.h>
#include <wait.h>
#include <stdlib.h>

int main() {
   int a = 0;
   pid_t val = fork();
   if(val == 0) {
        a += 1;
       printf("%d \n", a);
       exit(0);
   }
   if(val > 0){
       wait(NULL);
       printf("%d \n",a);
   }
}
