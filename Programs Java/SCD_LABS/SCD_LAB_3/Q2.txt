import java.util.Map;
import java.util.Scanner;
import java.util.HashMap;

public class Main{
    public static void main(String []args){
        System.out.println("Enter the Sample Text: ");
        HashMap<String, Integer> text = new HashMap<>();
        Scanner input = new Scanner(System.in);
        String data = input.nextLine();
        data = data.replaceAll("[^a-zA-Z0-9 ]","");
        String [] arrOfData = data.split(" ");
        for(String s: arrOfData){
            if(text.containsKey(s)){
                Integer a = text.get(s);
                a++;
                text.put(s, a);
            }
            else{
                text.put(s,1);
            }
        }
        System.out.println("Size of map is: " + text.size());

        for(Map.Entry<String, Integer> entry : text.entrySet()){
            System.out.println(entry.getKey() + " " + entry.getValue());
        }
    }
}
